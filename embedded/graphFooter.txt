Examples:

  # line chart
  printf "1 cat 3\\n3 cat 5\\n4 cat 4" \
  | tablegraph graph \
      --fieldsep ' ' \
      --type line

  # stacked bar time chart
  printf "date,type,lines\n2022,added,1\n2022,deleted,2\n2023,added,4" \
  | tablegraph graph \
      --fieldnames firstline \
      --type stacked-bar \
      --x-scale-type utc \
      --x-time-unit utcyear \
      --x-type temporal \
      --y-type quantitative
